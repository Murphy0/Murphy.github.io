LRU算法

算法 LRU
算法
发布日期:   2020-10-18

1、介绍

LRU（Least Recently Used）是一种常见的页面置换算法。

常见的页面置换算法有如下几种：

  • LRU 最近最久未使用
  • FIFO 先进先出置换算法 类似队列
  • OPT 最佳置换算法 （理想中存在的）
  • NRU Clock置换算法
  • LFU 最少使用置换算法
  • PBA 页面缓冲算法

2、原理

​ LRU的设计原理就是，当数据在最近一段时间经常被访问，那么它在以后也会经常被访问。这就意味着，如果经常访问的数据，我们需要然其能够快速命中，而不常访问的数据，我们在容量超出限制内，要将其淘汰。

3、代码实现

3.1、思路

  • 构建双向链表节点LRUNode，应包含key,value,prev,next这几个基本属性
  • 对于cache对象get操作，直接获取，成功则移位链表head.next 位置
  • 对于cache对象remove操作，直接移除，成功则构建当前节点前后节点关系
  • 对于cache对象put操作：
    • 先判断key是否存在，若存在则覆盖，且移位到头部
    • 若key不存在判断容量是否超载，超载则删除tail.pre节点
    • 构建新节点，移位链表head.next 位置

/**
 * @Author: Murphy
 * @Version 1.0
 */
public class LRUCache<K, V> {

    /**
     * 默认总容量256
     */
    private int capacity = 256;

    /**
     * 当前容量
     */
    private int currentSize;

    /**
     * Node记录
     */
    private Map<K, LRUNode<K, V>> cache = new ConcurrentHashMap<>(128);

    /**
     * 头节点
     */
    private final LRUNode<K, V> head;

    /**
     * 尾节点
     */
    private final LRUNode<K, V> tail;

    private AtomicBoolean status = new AtomicBoolean(false);


    public LRUCache(int capacity) {
        this();
        if (capacity < 0) {
            throw new RuntimeException("capacity illegal");
        }
        this.capacity = capacity;
    }

    public LRUCache() {
        //固定首/尾节点
        this.head = new LRUNode<>();
        this.tail = new LRUNode<>();
        this.head.next = this.tail;
        this.head.pre = null;
        this.tail.pre = this.head;
        this.tail.next = null;
    }

    /**
     * 获取
     *
     * @param key
     * @return
     */
    public V get(K key) {
        LRUNode<K, V> node = cache.get(key);
        if (node == null) {
            return null;
        }
        //移位到头节点
        moveNode2Head(node);

        return node.value;
    }
  
    /**
     * 移除
     *
     * @param key
     */
    public void remove(K key) {
        //判断key是否存在
        LRUNode<K, V> node = cache.get(key);
        if (node == null) {
            return;
        }
        //移除缓存
        cache.remove(key);
        currentSize--;
        //移出链表
        LRUNode<K, V> pre = node.pre;
        LRUNode<K, V> next = node.next;
        pre.next = next;
        next.pre = pre;
    }

    /**
     * 新增缓存
     *
     * @param key
     */
    public void cach(K key, V value) {
        //自旋
        while (!status.compareAndSet(false, true)) {
        }

        try {
            //判断key是否存在，若存在则覆盖，且移位到头部
            LRUNode<K, V> node = cache.get(key);
            if (node != null) {
                //覆盖，并移位头部
                node.value = value;
                moveNode2Head(node);
            } else {
                LRUNode<K, V> newNode = new LRUNode<>();
                newNode.key = key;
                newNode.value = value;
                //若key不存在判断容量是否超载
                if (currentSize >= capacity) {
                    //移除最后一个
                    remove(this.tail.pre.key);
                }
                //将此节点移位头部
                moveNode2Head(newNode);
                //缓存
                cache.put(key, newNode);
                currentSize++;
            }
        } finally {
            //操作完成恢复状态值
            status.getAndSet(Boolean.FALSE);
        }
    }

    private void moveNode2Head(LRUNode<K, V> node) {
        //如果当前节点就是头节点，则不变
        if (this.head.next == node) {
            return;
        }

        //如果是新节点,则直接放头部
        if (node.pre == null && node.next == null) {
            node.pre = this.head;
            node.next = this.head.next;
            this.head.next.pre = node;
            this.head.next = node;
            return;
        }

        //移位至头部
        LRUNode<K, V> pre = node.pre;
        LRUNode<K, V> next = node.next;

        pre.next = next;
        next.pre = pre;
        //设置头节点
        node.next = this.head.next;
        node.pre = this.head;

        this.head.next.pre = node;
        this.head.next = node;
    }

}
