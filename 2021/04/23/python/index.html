2.1、字符串

1、基本操作

name = "murphy"
# 截取下标为0的字符 结果m
print(name[0])
# 截取下标为[0,3)的字符 结果mur
print(name[0:3])
# 截取下标为[1,3)的字符 结果ur
print(name[1:3])
# 截取下标为[1,7)且步进值为2的字符 结果upy
print(name[1:7:2])
# 截取从下标2开始往后的字符 结果rphy
print(name[2:])
# 截取下标5以前的全部字符 结果murph
print(name[:5])
# \n转译
print("hello\nmurphy")
# 不再转译
print(r"hello\nmurphy")

2、常用API

2.2、列表(List)

  • 有序且存储无固定数据类型

1、插入

demoList = ["murphy",123,"francis"]
demoList2 = ["test",456,"tanhui"]
demoList3 = ["test3",3,"3"]
demoList4 = ["test4",4,"4"]
demoList5 = ["test5",5,"5"]
for i in demoList:
    print(i)
# 尾部追加元素
demoList.append("fish")
# 列表嵌套 结果['murphy', 123, 'francis', 'fish', ['test', 456, 'tanhui']]
demoList.append(demoList2)
# 列表数据聚合 结果['test', 456, 'tanhui', 'test3', 3, '3']
demoList2.extend(demoList3)
# 数据插入 在下标为1的地方插入insert1 结果为['test4', 'insert1', 4, '4']
demoList4.insert(1,"insert1")
# 数据插入 从后往前在下标为2的地方插入insert1 结果为['test5', 'insert1', 5, '5']
demoList5.insert(-2,"insert1")

2、删除

demoList = ["murphy",123,"francis"]
demoList2 = ["test",456,"tanhui"]
demoList3 = ["test3",3,"3"]
demoList4 = ["test4",4,"4",4,4]

# 删除指定下标数据，此时index不能超边界 结果['murphy', 'francis']
del demoList[1]
# 结果[456, 'tanhui']
del demoList2[-3]
# 弹出末尾最后一个['test3', 3]
demoList3.pop()
# 移除值,只删除找到的第一个值 结果['test4', '4', 4, 4]
demoList4.remove(4)

3、修改

demoList = ["murphy",123,"francis"]
demoList2 = ["test",456,"tanhui"]

# 指定下标修改 结果['and23', 123, 'francis']
demoList[0] = "and23"
# 结果['test', 456, 123]
demoList2[-1] = 123

4、查

demoList = ["murphy",123,"francis"]
demoList2 = ["a","c","d","e","f","c"]
demoList3 = [1,9,20,3,4,2,1,7]
name = "murphy"
# 判断元素是否在列表中 结果True
flag = name in demoList
# False
flag2 = name not in demoList
# 查找从下标1到下标3之间存在的"c"字符的下标位置，查找范围：[1,3) 结果：1  同时找不到会报错
index = demoList2.index("c",1,3)
# 计数 2
count = demoList2.count("c")
# 反转列表 [7, 1, 2, 4, 3, 20, 9, 1]
demoList3.reverse()
# 升序排序 [1, 1, 2, 3, 4, 7, 9, 20]
demoList3.sort()
# 降序排序[20, 9, 7, 4, 3, 2, 1, 1]
demoList3.sort(reverse=True)
# ['a', 'c', 'c', 'd', 'e', 'f']
demoList2.sort()

# 列表转化字典模式，打印下标和值 结果如下：
# index:0,value:1
# index:1,value:9
# index:2,value:20
# index:3,value:3
# index:4,value:4
# index:5,value:2
# index:6,value:1
# index:7,value:7
for index,value in enumerate(demoList3):
    print("index:%s,value:%s"%(index,value))

2.3、元组(Tuple )

  • 有序数据不可改变，但是可以包含可变对象；若tuple只有1个元素，必须逗号结尾
tup1=(2,"abd",123,2,"abd")
tup2=("abc",1)
# 结果2
print(tup1[0])
# 结果abd
print(tup1[-1])
# 结果 ('abd',)
print(tup1[1:2])
# ('abd', 2)
print(tup1[1:4:2])
# ('abd', 123)
print(tup1[-1:-4:-2])
# 增加 ('abc', 1, 2, 'abd', 123, 2, 'abd') 本质是新增一个元组来做合并操作
tup = tup2+tup1
print(tup)
# 删除元组,此处删除的是tup1 变量，之后的逻辑不可使用
del tup1

2.4、字典(dict)

  • 字典是无序的对象集合，使用key-value存储
  • key必须使用不可变类型
  • key唯一

1、增加

dict = {"name":"murphy","age":25}
# 结果 {'name': 'murphy', 'age': 25, 'sex': '男'}
dict["sex"] = "男"

2、删除

dict = {"name":"murphy","age":25}
dict2 = {"name2":"murphy2","age2":25}
# 结果 {'name': 'murphy', 'age': 25, 'sex': '男'}
dict["sex"] = "男"

# 此处删除的是整个sex键值对，后续如果再通过dict["sex"]访问会报错
# 若移除的key不存在会报错
del dict["sex"]
# 变量删除
del dict
# 移除指定key，等价于del dict["name2"]，若移除的key不存在会报错
dict2.pop("name2")
print(dict2)

# 清空字段 结果：{}
dict2.clear()

3、修改

dict = {"name":"murphy","age":25}
dict["age"] = 20

4、查

dict = {"name":"murphy","age":25}
# dict_keys(['name', 'age'])
print(dict.keys())
# dict_values(['murphy', 25])
print(dict.values())
# dict_items([('name', 'murphy'), ('age', 25)])
print(dict.items())
# 遍历key
for key in dict.keys():
    print(key)

# 遍历item
for key,value in dict.items():
    # 结果 key=name,value=murphy
    print("key=%s,value=%s"%(key,value))

3、函数

1、定义

def divid(a, b):
    sum = a + b
    yushu = a % b
    return sum, yushu


# 接收多返回值 此时r1结果为 (38, 10)
r1 = divid(10, 28)
# 结果re1:38,re2:10
print("re1:%s,re2:%s" % r1)

2、变量

a = 55

def method1():
    a = 100
    print(a)

def method2():
    print(a)

method1()
method2()
print(a)

# 结果
# 100
# 55
# 55

可见有局部变量的时候使用局部变量，没有的时候使用全局变量
a = 55

def method1():
    global a #此处表示引用全局变量到局部位置，之后的操作会修改全局变量
    a = 100
    print(a)

def method2():
    a = 320
    print(a)

method1()
method2()
print(a)

# 结果
# 100
# 320
# 100
